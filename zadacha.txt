#include <stdio.h>
#define N 4
int main() {
    int a[N][N];

    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            if (i == j)
                a[i][j] = 0;      
            else if (i < j)
                a[i][j] = 1;      
            else
                a[i][j] = 2;      
        }
    }

    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            printf("%d ", a[i][j]);
        }
        printf("\n");
    }

    return 0;
}
